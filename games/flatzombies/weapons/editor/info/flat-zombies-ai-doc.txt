=== СТРУКТУРА ПРОЕКТА ===

Мобильная 2D игра с видом сбоку на движке Unity Engine v6.4 от 2026 года + язык программирования C# v9.0.
Управление персонажем происходит с помощью кнопок Влево/Вправо.
Стрельба происходит нажатием на экран в любом месте. Персонаж получает угол наклона для рук и он стреляет в точку нажатия.
Моды на оружие имеют поддержку Lua-скрипта, который подписывается на события.
Lua работает с помощью плагина xLua от Tencent.
В Lua есть два способа обращения к методам объектам - для вызова нестатических методов C# объекта нужно использовать двоеточие (:)
Слушатели события добавляются при загрузке мода в игру в одном из классов, отвечающих за загрузку:

public WeaponModHandler(Weapon weapon, ModMetaData jsonData) {
	luaEnv = new LuaEnv();
	AllowOnlyTheseNamespaces(luaEnv, whiteList); //Проверка на безопасность
	luaEnv.DoString(jsonData.luaScript); //Выполняем Lua-скрипт
	//Получить ссылку на функции из Lua
	this.onStart = luaEnv.Global.GetInPath<LuaFunction>("onStart");
	this.onKeyShot = luaEnv.Global.GetInPath<LuaFunction>("onKeyShot");
	this.onShot = luaEnv.Global.GetInPath<LuaFunction>("onShot");
	this.onReloadStart = luaEnv.Global.GetInPath<LuaFunction>("onReloadStart");
	this.onReloadUpdate = luaEnv.Global.GetInPath<LuaFunction>("onReloadUpdate");
	this.onReloadComplete = luaEnv.Global.GetInPath<LuaFunction>("onReloadComplete");
	this.onBulletHit = luaEnv.Global.GetInPath<LuaFunction>("onBulletHit");
	this.onSwitchWeapon = luaEnv.Global.GetInPath<LuaFunction>("onSwitchWeapon");
	this.onSwitchEnable = luaEnv.Global.GetInPath<LuaFunction>("onSwitchEnable");
	this.onSwitchDisable = luaEnv.Global.GetInPath<LuaFunction>("onSwitchDisable");
	this.onDestroy = luaEnv.Global.GetInPath<LuaFunction>("onDestroy");
	//Подписываемся на события оружия и при срабатывании события мы вызываем Lua-функции 
	if (this.onStart != null) weapon.onStart.AddListener((Weapon weapon) => this.CallLuaListener(onStart, weapon));
	if (this.onKeyShot != null) weapon.onKeyShot.AddListener((Weapon weapon) => this.CallLuaListener(onKeyShot, weapon));
	if (this.onShot != null) weapon.onShot.AddListener((Weapon weapon) => this.CallLuaListener(onShot, weapon));
	if (this.onUpdate != null) weapon.onUpdate.AddListener((Weapon weapon) => { this.CallLuaListener(onUpdate, weapon); weapon.updateHasListeners = true; });
	if (this.onReloadStart != null) weapon.onReloadStart.AddListener((Weapon weapon) => this.CallLuaListener(onReloadStart, weapon));
	if (this.onReloadUpdate != null) weapon.onReloadUpdate.AddListener((Weapon weapon) => this.CallLuaListener(onReloadUpdate, weapon));
	if (this.onReloadComplete != null) weapon.onReloadComplete.AddListener((Weapon weapon) => this.CallLuaListener(onReloadComplete, weapon));
	if (this.onBulletHit != null) weapon.onBulletHit.AddListener((BulletHit bulletHit) => this.CallLuaListener(onBulletHit, bulletHit));
	if (this.onSwitchWeapon != null) weapon.onSwitchWeapon.AddListener((Weapon weapon, bool show) => this.CallLuaListener(onSwitchWeapon, weapon, show));
	if (this.onSwitchEnable != null) weapon.onSwitchEnable.AddListener((Weapon weapon) => this.CallLuaListener(onSwitchEnable, weapon));
	if (this.onSwitchDisable != null) weapon.onSwitchDisable.AddListener((Weapon weapon) => this.CallLuaListener(onSwitchDisable, weapon));
	if (this.onDestroy != null) weapon.onDestroy.AddListener((Weapon weapon) => this.CallLuaListener(onDestroy, weapon));
}

function onStart(weapon)

function onShot(weapon)

function onReloadStart(weapon)

Как загружается мод, возможности загрузчика и JSON-парсера:
Мод изначально представлен в виде плоского JSON-файла ("object.property.data.name": "value")
Мод содержит идентификатор префаба, который используется как родитеский базовый префаб.
Для загрузки мода в руки персонажа создаём префаб имеющийся в игре и вносим изменения с помощью фукнции JsonToObject():
Weapon weapon = Weapon.Instantiate(...);
JsonParser.JsonToObject(weapon, jsonFullText, weapon.id, "weapon.") //Последнее значение "weapon." указывает, что мы будет искать все ключи, которые начинаются с "weapon."
Алгоритм работы парсера: 
Есть строка JSON: "weapon.magazine.SpriteRenderer.sprite": "data:image/png;base64..."
В процессе чтения этой строки, парсер ищет "magazine" - сначала мы пытаемся найти компонент MonoBehaviour с таким именем, затем мы перебираем все дочерние объекты с таким именем.
В итоге мы находим дочерний объект с именем "magazine". Внутри него мы ищем "SpriteRenderer", мы находим компонент<SpriteRenderer> и внутри него ищем свойство sprite.
В результате мы получаем доступ к спрайту на дочернем объект и меняем этот спрайт. 
Строка в json имеет такой формат: родитель.[дочернийОбъект/компонент/свойство].[дочернийОбъект/компонент/свойство]: [значение]
Для изменения элемента массива можно указывать индекс элемента: "weapon.objectList[0].sounds[1].audioClip": "data:audio/wav;base64..." 
Можно использовать вызов методов для получения ссылки на другой объект и вносить изменения в его свойства: "weapon.GetComponents<Collider2D>()[1].enabled": false - получаем массив и отключить второй Collider2D.
Парсер умеет проходить путь к свойству не только по обычным публичным полям и свойствам, но и по компонентам с указанием их типа и по элементам массива с указанаем числового индекса, как было указано выше.
Родительский объект указан при вызове JsonToObject()
Для создания мода используется отдельная программа-редактор.
В редакторе мода можно указать дополнительные поля в JSON, например, для добавления новых компонентов MonoBehaviour в дочерние объекты на сцене у оружия.
json["removedComponents"] - список компонентов для удаления, должен иметь имя дочернего объекта и тип компонента: "child.SpriteRenderer, otherChild.Collider2D", несколько штук через запятую, здесь child и otherChild являются дочерними объектами к оружию Weapon
json["removedGameObjects"] - имена GameObject для удаления: "child.other, otherChild", несколько штук через запятую, здесь child и otherChild являются дочерними объектами к оружию Weapon
json["addedGameObjects"] - Добавить дочерние объекты, несколько штук через запятую
json["addedComponents"] - Добавить компоненты, несколько штук через запятую, строки в формате "child.SpriteRenderer, otherChild.Collider2D", здесь child и otherChild являются дочерними объектами к оружию Weapon
С помощью этих ключей из массива мы можем создавать дополнительные компоненеты и объекты для мода, для объекта Weapon на сцене.
Если к моду нужно добавить текстуры, звуки и другие ассеты, то это можно делать с помощью создания компонентов MonoBehaviour к GameObject оружия и к его дочерним объектам.
Как работает редактор:
Все ключи из json отображаются в виде списка и у каждого свойства есть поле для изменения значения. 
Если значение хранит координаты x и y, то на экране отображаются два поля ввода для x и для y.
Если значение хранит массив спрайтов/текстур, то на экране отображается список текстовых полей в которых текстура записана как base64 и рядом кнопка Заменить, при нажатии на неё появляется файловый менеджер для загрузки другого файла.
Редактор имеет отдельный список с мета-данными для всех основных свойств оружия. 
В списке с мета-данными указан путь свойства, его тип и его описание, например часть списка:

Мы можем добавить пользовательские свойства, которые ранее не были нигде указаны в мета-данных редактора, но есть в объектах у Unity Engine и в игровых классах.
Редактор имеет на экране кнопку "Добавить параметр", который показывает поле для добавления нового свойства.
Редактор имеет HTML-форму для добавления нового свойства:

Таким образом мы можем добавлять новые ассеты, если это нужно для Lua-скриптов
Для некоторых типов необязательно указывать значение, оно будет доступно для редактирования после нажатия Добавить, в общем списке полявится для него своё поле для редактирования.
Если нужно добавлить текстуру или звук для Lua-скрипта, то напиши об этом в виде инструкции с пунктами для простого пользователя: нажми добавить новый параметр, укажи путь свойства и тип свойства.
Для добавления ассета/текстуры нужен объект-компонент подключенный к GameObject, который будет хранить в себе этот ассет.

Редактор так же имеет кнопку Добавить объект. 
При нажатии Добавить объект появляется список для выбора компонента, который будет добавлен к объекту: 
Transform, AnimatorSprite, SpriteRenderer.
К ним так же можно добавить текстуры для Lua-скрипта.
Выбираем компонент и нужно указать имя объекта. 
Имя объекта и его компонент автоматически появятся в списках addedGameObjects и addedComponents - об этом можно не говорить пользователю, он это увидит сам.
Для редактирования нового добавленного Unity-компонента следует добавить пользовательские свойства через кнопку "Добавить параметр", в пути свойства мы указываем имя нового объекта и тип данных.
Если надо добавить компонент класса AnimatorSprite, то нужно нажать Добавить объект, затем выбрать AnimatorSprite, ввести имя объекта и нажать Подтвердить.
В редакторе появится свойство "объект.AnimatorSprite.animations", которое содержит анимации и список спрайтов.
При экспорте оно будет записано в формате JSON.
